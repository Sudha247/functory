\begin{frame}\frametitle{Interface}

  \begin{ocaml}
  val compute : 
    worker:('a -> 'b) -> 
    master:('a * 'c -> 'b -> ('a * 'c) list) -> 
    ('a * 'c) list -> 
    unit
  \end{ocaml}



  \begin{itemize}
  \item \ocaml{'a * 'c} représente une tâche
  \item son résultat est de type \ocaml{'b}
  \item une tâche terminée peut produire de nouvelles tâches
  \item le calcul termine quand il n'y a plus de tâches
  \end{itemize}
\end{frame}

\begin{frame}\frametitle{Réseau}
  utilise
  \begin{itemize}
  \item une architecture client/serveur TCP/IP
  \item la sérialisation d'Ocaml (\textit{marshaling})
  \end{itemize}
  
  \vfill
  trois cas de figure
  \begin{enumerate}
  \item \emph{même binaire} : on peut sérialiser les fonctions
  \item \emph{même version d'Ocaml} : on peut seulement sérialiser les valeurs
  \item \emph{sinon} : on ne peut sérialiser que les chaînes
  \end{enumerate}
\end{frame}

\begin{frame}\frametitle{Trois modules pour le réseau}
\emph{Same}
\begin{ocaml}
val compute : (* same as before *) ...
\end{ocaml}



\emph{Poly}
    \begin{ocaml}
val Worker.compute : ('a -> 'b) -> unit
val Master.compute : 
  ('a * 'c -> 'b -> ('a * 'c) list) ->
  ('a * 'c) list -> unit
\end{ocaml}



\emph{Mono}
    \begin{ocaml}
val Worker.compute : (string -> string) -> unit
val Master.compute : 
  (string * 'c -> string -> (string * 'c) list) ->
  (string * 'c) list -> unit
\end{ocaml}
\end{frame}

\begin{frame}\frametitle{Protocole}
  \only<1>{le patron envoie une tâche à un ouvrier}
  \only<2>{l'ouvrier calcule et renvoie le résultat}
  \only<3-4>{patron et ouvriers échanges des messages \emph{ping}/\emph{pong}}
  \only<5>{la patron envoie une autre tâche à l'ouvrier}
  \only<6>{en cas de déconnexion...}
  \only<7>{la tâche est \emph{reprogrammée} sur un autre ouvrier}
  \only<8>{quand un ouvrier a terminé son calcul...}
  \only<9>{l'autre est interrompu}
  \only<10>{le patron est prévenu dès qu'un calcul échoue}
  \only<11>{à la fin, le patron peut demander aux ouvriers de terminer}

\begin{frame}\frametitle{Interface dérivée}

quelques fonctions pratiques sont fournies \par
elles sont dérivées de l'interface générique


\small
\begin{ocaml}
val map : ('a -> 'b) -> 'a list -> 'b list

val map_{local,remote}_fold : 
  f:('a -> 'b) -> fold:('c -> 'b -> 'c) -> 'c -> 'a list -> 'c   

val map_fold_{a,ac} : 
  f:('a -> 'b) -> fold:('b -> 'b -> 'b) -> 'b -> 'a list -> 'b 
\end{ocaml}
\end{frame}

\begin{frame}\frametitle{Exemple}
  \begin{displaymath}
    \sum_{0\le n\le 100}f(n)
  \end{displaymath}


  
\begin{ocaml}
open Functory.Cores
let () = set_number_of_cores 4

    
let f x = ...
    
let rec list n = if n = 0 then [] else n :: list (n-1)

let r = map_local_fold ~f ~fold:(+) 0 (list 100)
\end{ocaml}
\end{frame}

\begin{frame}\frametitle{Exemple}
  \begin{displaymath}
    \sum_{0\le n\le 100}f(n)
  \end{displaymath}


  
\begin{ocaml}
open Functory.Network
let () = declare_workers ~n:4 "moloch"
open Same    

let f x = ...
    
let rec list n = if n = 0 then [] else n :: list (n-1)

let r = map_local_fold ~f ~fold:(+) 0 (list 100)
\end{ocaml}
\end{frame}

\begin{frame}\frametitle{Expérimentations}
  exemple initial
  \begin{itemize}
  \item 80 obligations de preuve
  \item 4 démonstrateurs
  \item une minute maximum par obligation
  \item réseau de 3 machines (4, 8 et 8 c\oe urs)
  \end{itemize}


  \begin{itemize}
  \item calcul séquentiel \emph{$>$ 6 heures}
  \item avec Functory : \emph{22,5 minutes}
  \item accélération = \emph{16} (optimal = 20)
  \end{itemize}
